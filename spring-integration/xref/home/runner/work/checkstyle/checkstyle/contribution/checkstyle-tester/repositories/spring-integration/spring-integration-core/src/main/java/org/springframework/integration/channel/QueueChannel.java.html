17  package org.springframework.integration.channel;
18  
19  import java.util.ArrayList;
20  import java.util.List;
21  import java.util.Queue;
22  import java.util.concurrent.BlockingQueue;
23  import java.util.concurrent.LinkedBlockingQueue;
24  import java.util.concurrent.Semaphore;
25  import java.util.concurrent.TimeUnit;
26  
27  import org.springframework.integration.core.MessageSelector;
28  import org.springframework.integration.support.management.metrics.GaugeFacade;
29  import org.springframework.integration.support.management.metrics.MetricsCaptor;
30  import org.springframework.lang.Nullable;
31  import org.springframework.messaging.Message;
32  import org.springframework.util.Assert;
33  
34  /**
35   * Simple implementation of a message channel. Each {@link Message} is placed in
36   * a {@link BlockingQueue} whose capacity may be specified upon construction.
37   * The capacity must be a positive integer value. For a zero-capacity version
38   * based upon a {@link java.util.concurrent.SynchronousQueue}, consider the
39   * {@link RendezvousChannel}.
40   *
41   * @author Mark Fisher
42   * @author Oleg Zhurakousky
43   * @author Gary Russell
44   * @author Artem Bilan
45   */
46  public class QueueChannel extends AbstractPollableChannel implements QueueChannelOperations {
47  
48  	private final Queue<Message<?>> queue;
49  
50  	protected final Semaphore queueSemaphore = new Semaphore(0); // NOSONAR final
51  
52  	@Nullable
53  	private GaugeFacade sizeGauge;
54  
55  	@Nullable
56  	private GaugeFacade remainingCapacityGauge;
57  
58  	/**
59  	 * Create a channel with the specified queue.
60  	 *
61  	 * @param queue The queue.
62  	 */
63  	public QueueChannel(Queue<Message<?>> queue) {
64  		Assert.notNull(queue, "'queue' must not be null");
65  		this.queue = queue;
66  	}
67  
68  	/**
69  	 * Create a channel with the specified queue capacity.
70  	 *
71  	 * @param capacity The capacity.
72  	 */
73  	public QueueChannel(int capacity) {
74  		Assert.isTrue(capacity > 0, "The capacity must be a positive integer. " +
75  				"For a zero-capacity alternative, consider using a 'RendezvousChannel'.");
76  		this.queue = new LinkedBlockingQueue<>(capacity);
77  	}
78  
79  	/**
80  	 * Create a channel with "unbounded" queue capacity. The actual capacity value is
81  	 * {@link Integer#MAX_VALUE}. Note that a bounded queue is recommended, since an
82  	 * unbounded queue may lead to OutOfMemoryErrors.
83  	 */
84  	public QueueChannel() {
85  		this(new LinkedBlockingQueue<>());
86  	}
87  
88  	@Override
89  	public void registerMetricsCaptor(MetricsCaptor metricsCaptor) {
90  		super.registerMetricsCaptor(metricsCaptor);
91  		this.sizeGauge =
92  				metricsCaptor.gaugeBuilder("spring.integration.channel.queue.size", this,
93  						(channel) -> getQueueSize())
94  						.tag("name", getComponentName() == null ? "unknown" : getComponentName())
95  						.tag("type", "channel")
96  						.description("The size of the queue channel")
97  						.build();
98  
99  		this.remainingCapacityGauge =
100 				metricsCaptor.gaugeBuilder("spring.integration.channel.queue.remaining.capacity", this,
101 						(channel) -> getRemainingCapacity())
102 						.tag("name", getComponentName() == null ? "unknown" : getComponentName())
103 						.tag("type", "channel")
104 						.description("The remaining capacity of the queue channel")
105 						.build();
106 	}
107 
108 	@Override
109 	protected boolean doSend(Message<?> message, long timeout) {
110 		Assert.notNull(message, "'message' must not be null");
111 		try {
112 			if (this.queue instanceof BlockingQueue) {
113 				BlockingQueue<Message<?>> blockingQueue = (BlockingQueue<Message<?>>) this.queue;
114 				if (timeout > 0) {
115 					return blockingQueue.offer(message, timeout, TimeUnit.MILLISECONDS);
116 				}
117 				if (timeout == 0) {
118 					return blockingQueue.offer(message);
119 				}
120 				blockingQueue.put(message);
121 				return true;
122 			}
123 			else {
124 				try {
125 					return this.queue.offer(message);
126 				}
127 				finally {
128 					this.queueSemaphore.release();
129 				}
130 			}
131 		}
132 		catch (InterruptedException e) {
133 			Thread.currentThread().interrupt();
134 			return false;
135 		}
136 	}
137 
138 	@Override
139 	@Nullable
140 	protected Message<?> doReceive(long timeout) {
141 		try {
142 			if (timeout > 0) {
143 				if (this.queue instanceof BlockingQueue) {
144 					return ((BlockingQueue<Message<?>>) this.queue).poll(timeout, TimeUnit.MILLISECONDS);
145 				}
146 				else {
147 					return pollNonBlockingQueue(timeout);
148 				}
149 			}
150 			if (timeout == 0) {
151 				return this.queue.poll();
152 			}
153 
154 			if (this.queue instanceof BlockingQueue) {
155 				return ((BlockingQueue<Message<?>>) this.queue).take();
156 			}
157 			else {
158 				Message<?> message = this.queue.poll();
159 				while (message == null) {
160 					this.queueSemaphore.tryAcquire(50, TimeUnit.MILLISECONDS); // NOSONAR ok to ignore result
161 					message = this.queue.poll();
162 				}
163 				return message;
164 			}
165 		}
166 		catch (InterruptedException e) {
167 			Thread.currentThread().interrupt();
168 			return null;
169 		}
170 	}
171 
172 	@Nullable
173 	private Message<?> pollNonBlockingQueue(long timeout) throws InterruptedException {
174 		Message<?> message = this.queue.poll();
175 		if (message == null) {
176 			long nanos = TimeUnit.MILLISECONDS.toNanos(timeout);
177 			long deadline = System.nanoTime() + nanos;
178 			while (message == null && nanos > 0) {
179 				this.queueSemaphore.tryAcquire(nanos, TimeUnit.NANOSECONDS); // NOSONAR ok to ignore result
180 				message = this.queue.poll();
181 				if (message == null) {
182 					nanos = deadline - System.nanoTime();
183 				}
184 			}
185 		}
186 		return message;
187 	}
188 
189 	@Override
190 	public List<Message<?>> clear() {
191 		List<Message<?>> clearedMessages = new ArrayList<>();
192 		if (this.queue instanceof BlockingQueue) {
193 			((BlockingQueue<Message<?>>) this.queue).drainTo(clearedMessages);
194 		}
195 		else {
196 			Message<?> message;
197 			while ((message = this.queue.poll()) != null) {
198 				clearedMessages.add(message);
199 			}
200 		}
201 		return clearedMessages;
202 	}
203 
204 	@Override
205 	public List<Message<?>> purge(@Nullable MessageSelector selector) {
206 		if (selector == null) {
207 			return this.clear();
208 		}
209 		List<Message<?>> purgedMessages = new ArrayList<>();
210 		Object[] array = this.queue.toArray();
211 		for (Object o : array) {
212 			Message<?> message = (Message<?>) o;
213 			if (!selector.accept(message) && this.queue.remove(message)) {
214 				purgedMessages.add(message);
215 			}
216 		}
217 		return purgedMessages;
218 	}
219 
220 	@Override
221 	public int getQueueSize() {
222 		return this.queue.size();
223 	}
224 
225 	@Override
226 	public int getRemainingCapacity() {
227 		if (this.queue instanceof BlockingQueue) {
228 			return ((BlockingQueue<Message<?>>) this.queue).remainingCapacity();
229 		}
230 		else {
231 			//Assume that underlying Queue implementation takes care of its size on "offer".
232 			return Integer.MAX_VALUE;
233 		}
234 	}
235 
236 	@Override
237 	public void destroy() {
238 		super.destroy();
239 		if (this.sizeGauge != null) {
240 			this.sizeGauge.remove();
241 		}
242 		if (this.remainingCapacityGauge != null) {
243 			this.remainingCapacityGauge.remove();
244 		}
245 	}
